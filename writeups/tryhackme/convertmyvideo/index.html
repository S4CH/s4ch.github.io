  $("#convert").click(function () {
    $("#message").html("Converting...");
    $.post(
      "/",
      { yt_url: "https://www.youtube.com/watch?v=" + $("#ytid").val() },
      function (data) {
        try {
          data = JSON.parse(data);
          if (data.status == "0") {
            $("#message").html(
              "<a href='" + data.result_url + "'>Download MP3</a>"
            );
          } else {
            console.log(data);
            $("#message").html("Oops! something went wrong");
          }
        } catch (error) {
          console.log(data);
          $("#message").html("Oops! something went wrong");
        }
      }
    );
  });
});
